## Why Functional Tests Are Not Enough

Your functional tests pass. Every button works, every form submits, every API returns the right data. But a CSS change just moved the checkout button off-screen on mobile. No functional test catches that. Visual regression testing does.

## The Generated Visual Test Suite

```typescript
// e2e/visual/pages.spec.ts
import { test, expect } from '@playwright/test';
import { maskDynamicContent, waitForStableDOM } from './helpers';

test.describe('Visual Regression — Desktop', () => {
  test.use({ viewport: { width: 1280, height: 720 } });

  test('homepage', async ({ page }) => {
    await page.goto('/');
    await waitForStableDOM(page);
    
    await expect(page).toHaveScreenshot('homepage-desktop.png', {
      maxDiffPixelRatio: 0.001,
    });
  });

  test('login page', async ({ page }) => {
    await page.goto('/login');
    await waitForStableDOM(page);
    
    await expect(page).toHaveScreenshot('login-desktop.png', {
      maxDiffPixelRatio: 0.001,
    });
  });

  test('dashboard with data', async ({ page }) => {
    // Login first
    await page.goto('/login');
    await page.fill('[data-testid="email"]', 'test@example.com');
    await page.fill('[data-testid="password"]', 'testpass123');
    await page.click('[data-testid="login-button"]');
    await page.waitForURL('/dashboard');
    
    // Mask dynamic content
    await maskDynamicContent(page);
    await waitForStableDOM(page);
    
    await expect(page).toHaveScreenshot('dashboard-desktop.png', {

        page.locator('[data-testid="current-time"]'),
        page.locator('[data-testid="user-avatar"]'),
        page.locator('[data-testid="activity-feed"]'),
      ],
    });
  });
```

## Handling Dynamic Content

The biggest challenge in visual testing is content that changes every time:

```typescript
// e2e/visual/helpers.ts
import { type Page } from '@playwright/test';

export async function waitForStableDOM(page: Page) {
  // Disable CSS animations
  await page.addStyleTag({
    content: `
      *, *::before, *::after {
        animation-duration: 0s !important;
        animation-delay: 0s !important;
        transition-duration: 0s !important;
        transition-delay: 0s !important;
      }
    `,
  });

  // Wait for network idle (all API calls resolved)
  await page.waitForLoadState('networkidle');
  
  // Wait for fonts to load
  await page.evaluate(() => document.fonts.ready);
  
  // Small buffer for any remaining rendering
  await page.waitForTimeout(500);
}

export async function maskDynamicContent(page: Page) {
  // Replace all timestamps with a fixed value
  await page.evaluate(() => {
    document.querySelectorAll('[data-testid*="date"], time').forEach((el) => {
      el.textContent = '2025-01-01';
    });
  });
  
  // Replace relative times ("3 minutes ago")
  await page.evaluate(() => {
    const walker = document.createTreeWalker(
      document.body,
      NodeFilter.SHOW_TEXT
    );
    while (walker.nextNode()) {
      const node = walker.currentNode;
      if (node.textContent?.match(/\d+ (minute|hour|day|week)s? ago/)) {
        node.textContent = '1 day ago';
      }
    }
  });
}

export async function screenshotWithViewports(
  page: Page,
  name: string,
  viewports: Array<{ width: number; height: number; label: string }>
) {
  for (const viewport of viewports) {
    await page.setViewportSize({ 
      width: viewport.width, 
      height: viewport.height 
    });
    await waitForStableDOM(page);
    
    // Playwright auto-manages baselines per viewport
    await page.screenshot({ 
      path: `screenshots/${name}-${viewport.label}.png`,
      fullPage: true 
    });
  }
}
```

## Key Takeaway

Visual regression testing catches the bugs that functional tests cannot see — layout shifts, missing elements, color changes, and responsive breakages. Set up Playwright visual tests with consistent environments (fixed viewport, disabled animations, masked dynamic content), test at multiple viewports for responsive coverage, and run them in CI with Docker for consistent rendering. The key to maintaining visual tests is baseline discipline: only update baselines for intentional changes, and always review diffs before approving. Claude can help set up the entire infrastructure, generate component-level visual tests, and even analyze visual diff images when tests fail.
